--- Ejercicio 1
---- Find the title of each film

FROM movies;

--- Ejercicio 2
---- Find the director of each film
SELECT Director
FROM movies;

--- Ejercicio 3
---- Find the title and director of each film
SELECT Director, Title
FROM movies;

--- Ejercicio 4
---- Find the title and year of each film
SELECT Title, Year
FROM movies;

--- Ejercicio 5
---- Find all the information of each film

FROM movies;
